86 //! Repeats Linduino loop
87 //! @return void
88 void loop()
89 {
90  uint8_t user_command;
91  uint8_t *addresses = NULL;
92  struct LT_2978FaultLog::FaultLogLtc2978 *ltc2978_fault_log;
93 
94 
95  if (Serial.available()) //! Checks for user input
96  {
97  while ((user_command = read_int()) == 0); //! Reads the user command
98  if (user_command != 'm')
99  Serial.println(user_command);
100 
101  uint8_t status;
102 
103  switch (user_command) //! Prints the appropriate submenu
104  {
105  case 1:
106  status = pmbus->readMfrFaultLogStatusByte(ltc2978_i2c_address);
107  if (status & LTC2978_SFL_EEPROM)
108  {
109  faultLog2978->read(ltc2978_i2c_address);
110 
111  faultLog2978->dumpBinary(&Serial);
112  Serial.println();
113  faultLog2978->print(&Serial);
114 
115  faultLog2978->release();
116  }
117  else
118  Serial.println(F("No LTC2978 Fault Log"));
119 
120  break;
121  case 2:
122  faultLog2978->clearFaultLog(ltc2978_i2c_address);
123 
124  break;
125  case 3:
126  pmbus->clearAllFaults(ltc2978_i2c_address);
127 
128  break;
129  case 4:
130  pmbus->enablePec(ltc2978_i2c_address);
131  delete faultLog2978;
132  delete smbus;
133  delete pmbus;
134  smbus = new LT_SMBusPec();
135  pmbus = new LT_PMBus(smbus);
136  faultLog2978 = new LT_2978FaultLog(pmbus);
137  break;
138  case 5:
139  pmbus->disablePec(ltc2978_i2c_address);
140  delete faultLog2978;
141  delete smbus;
142  delete pmbus;
143  smbus = new LT_SMBusNoPec();
144  pmbus = new LT_PMBus(smbus);
145  faultLog2978 = new LT_2978FaultLog(pmbus);
146  break;
147  case 6:
148  addresses = smbus->probe(0);
149  while (*addresses != 0)
150  {
151  Serial.print(F("ADDR 0x"));
152  Serial.println(*addresses++, HEX);
153  }
154  break;
155  case 7:
156  pmbus->restoreFromNvmGlobal();
157  delay(2000);
158  pmbus->resetGlobal();
159  break;
160  case 8:
161  smbus->sendByte(ltc2978_i2c_address, MFR_FAULT_LOG_STORE);
162  break;
163  default:
164  Serial.println(F("Incorrect Option"));
165  break;
166  }
167  print_prompt();
168  }
169 }
