103 //! Main Linduino loop
104 //! @return void
105 void loop()
106 {
107  uint8_t user_command;
108  uint16_t return_val;
109 
110  int i = 0;
111 
112  if (Serial.available()) //! Checks for user input
113  {
114  user_command = read_int(); //! Reads the user command
115  if (user_command != 'm')
116  Serial.println(user_command);
117 
118  switch (user_command) //! Prints the appropriate submenu
119  {
120 
121  case 1 :
122  Serial.print(F("\n****INITIALIZING THE LTC2970****\n"));
123  ltc2970_configure();
124  break;
125 
126  case 2 :
127  Serial.print(F("\n****ENABLE LTC2970 CHANNEL 0 AND CHANNEL 1****\n"));
128  ltc2970_dac_disconnect(smbus, ltc2970_i2c_address, 0);
129  ltc2970_gpio_up(smbus, ltc2970_i2c_address, 0);
130 
131  ltc2970_dac_disconnect(smbus, ltc2970_i2c_address, 1);
132  ltc2970_gpio_up(smbus, ltc2970_i2c_address, 1);
133  break;
134 
135  case 3 :
136  Serial.print(F("\n****SOFT CONNECT LTC2970 DAC0 and DAC1****\n"));
137  ltc2970_soft_connect_dac(smbus, ltc2970_i2c_address, 0);
138  ltc2970_soft_connect_dac(smbus, ltc2970_i2c_address, 1);
139  break;
140 
141  case 4 :
142  Serial.print(F("\n****SERVO CHANNEL 0 and 1 VOLTAGES 10% LOW****\n"));
143  ltc2970_servo_to_adc_val(smbus, ltc2970_i2c_address, 0, servo0_value_marg);
144  ltc2970_servo_to_adc_val(smbus, ltc2970_i2c_address, 1, servo1_value_marg);
145  break;
146 
147  case 5 :
148  Serial.print(F("\n****SERVO CHANNEL 0 and 1 VOLTAGES TO NOMINAL****\n"));
149  ltc2970_servo_to_adc_val(smbus, ltc2970_i2c_address, 0, servo0_value_nom);
150  ltc2970_servo_to_adc_val(smbus, ltc2970_i2c_address, 1, servo1_value_nom);
151  break;
152 
153  case 6 :
154  Serial.print(F("\n****ADC CH_0 VOLTAGE = (HEX VALUE)\n"));
155  return_val = smbus->readWord(ltc2970_i2c_address, LTC2970_CH0_A_ADC);
156  Serial.println(((return_val & 0x7FFF)*500e-6), DEC);
157  Serial.println(return_val, HEX);
158 
159  Serial.print(F("\n****ADC CH_1 VOLTAGE = (HEX VALUE)\n"));
160  return_val = smbus->readWord(ltc2970_i2c_address, LTC2970_CH1_A_ADC);
161  Serial.println(((return_val & 0x7FFF)*500e-6), DEC);
162  Serial.println(return_val, HEX);
163  break;
164 
165  case 7 :
166  Serial.print(F("\n****ADC CH_0 CURRENT = (HEX VALUE)\n"));
167  return_val = smbus->readWord(ltc2970_i2c_address, LTC2970_CH0_B_ADC);
168  Serial.println((((return_val & 0x7FFF)*500e-6)/0.007), DEC);
169  Serial.println(return_val, HEX);
170 
171  Serial.print(F("\n****ADC CH_1 CURRENT = (HEX VALUE)\n"));
172  return_val = smbus->readWord(ltc2970_i2c_address, LTC2970_CH1_B_ADC);
173  Serial.println((((return_val & 0x7FFF)*500e-6)/0.008), DEC);
174  Serial.println(return_val, HEX);
175  break;
176 
177  case 8 :
178  Serial.print(F("\n****PRINT FAULTS, CLEAR LATCHED FAULTS \n"));
179  ltc2970_read_faults(smbus, ltc2970_i2c_address);
180  break;
181 
182  case 9 :
183  Serial.print(F("\n****PRINT DIE TEMPERATURE \n"));
184  ltc2970_print_die_temp (smbus, ltc2970_i2c_address);
185  break;
186 
187  default:
188  Serial.println(F("Incorrect Option"));
189  break;
190  }
191  print_prompt();
192  }

232 //! Writes configuration values to the LTC2970 registers
233 //! @return void
234 void ltc2970_configure()
235 {
236  uint16_t return_val;
237  //start the 2970 by configuring all of its registers for this application
238  // use SMbus commands
239  smbus->writeWord(ltc2970_i2c_address, LTC2970_FAULT_EN, 0x0168);
240  smbus->writeWord(ltc2970_i2c_address, LTC2970_IO, 0x000A);
241  smbus->writeWord(ltc2970_i2c_address, LTC2970_ADC_MON, 0x007F);
242  smbus->writeWord(ltc2970_i2c_address, LTC2970_VDD_OV, 0x2CEC);
243  smbus->writeWord(ltc2970_i2c_address, LTC2970_VDD_UV, 0x2328);
244  smbus->writeWord(ltc2970_i2c_address, LTC2970_V12_OV, 0x3FFF);
245  smbus->writeWord(ltc2970_i2c_address, LTC2970_V12_UV, 0x00000);
246 
247  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH0_A_OV, 0x2AF8);
248  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH0_A_UV, 0x2328);
249  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH0_A_SERVO, 0x0000);
250  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH0_A_IDAC, 0x0084);
251  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH0_B_OV, 0x3FFF);
252  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH0_B_UV, 0x0000);
253 
254  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH1_A_OV, 0x1C5D);
255  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH1_A_UV, 0x1770);
256  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH1_A_SERVO, 0x0000);
257  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH1_A_IDAC, 0x0084);
258  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH1_B_OV, 0x3FFF);
259  smbus->writeWord(ltc2970_i2c_address, LTC2970_CH1_B_UV, 0x0000);
260 }
