70 int8_t LTC2668_write(uint8_t cs, uint8_t dac_command, uint8_t dac_address, uint16_t dac_code)
71 // Write the 16-bit dac_code to the LTC2668
72 {
73  static uint8_t last_data_array[4];
74  uint8_t data_array[4], rx_array[4];
75  int8_t ret;
76  LT_union_int16_2bytes data;
77 
78  data.LT_int16 = dac_code; // Copy DAC code to union
79  data_array[3] = 0; // Only required for 32 byte readback transaction
80  data_array[2] = dac_command | dac_address; // Build command / address byte
81  data_array[1] = data.LT_byte[1]; // MS Byte
82  data_array[0] = data.LT_byte[0]; // LS Byte
83 
84  spi_transfer_block(cs, data_array, rx_array, (uint8_t) 4);
85 
86  // Compare data read back to data that was sent the previous time this function was called
87  if ((rx_array[2] == last_data_array[2]) && (rx_array[1] == last_data_array[1]) && (rx_array[0] == last_data_array[0]))
88  {
89  ret = 0;
90  }
91  else
92  {
93  ret = 1;
94  }
95 
96  last_data_array[0] = data_array[0]; // Copy data array to a static array to compare
97  last_data_array[1] = data_array[1]; // the next time the function is called
98  last_data_array[2] = data_array[2];
99 
100  return(ret);
101 }

103 uint16_t LTC2668_voltage_to_code(float dac_voltage, float min_output, float max_output)
104 // Calculate a LTC2668 DAC code given the desired output voltage and the minimum / maximum
105 // outputs for a given softspan range.
106 {
107  uint16_t dac_code;
108  float float_code;
109  float_code = 65535.0 * (dac_voltage - min_output) / (max_output - min_output); // Calculate the DAC code
110  float_code = (float_code > (floor(float_code) + 0.5)) ? ceil(float_code) : floor(float_code); // Round
111  if (float_code < 0.0) float_code = 0.0;
112  if (float_code > 65535.0) float_code = 65535.0;
113  dac_code = (uint16_t) (float_code); // Convert to unsigned integer
114  return (dac_code);
115 }
