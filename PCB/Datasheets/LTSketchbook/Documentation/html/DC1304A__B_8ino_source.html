154 //! Set filter settings
155 //! @return void
156 void menu_1_filter_settings()
157 {
158  int16_t user_command; // The user input command
159  while (1)
160  {
161  Serial.print(F("*************************\n\n"));
162  Serial.println (F("0-Enter External Frequency"));
163  Serial.println (F("1-Set Gain"));
164  Serial.println (F("2-Set Lowpass"));
165  Serial.println (F("3-Set No Function Code"));
166  Serial.println (F("4-Upload Filter Settings"));
167  Serial.print (F("m-Main Menu\n"));
168  Serial.print (F("\nEnter a Command: "));
169  user_command = read_int(); // Read the single command
170  if (user_command == 'm')
171  return;
172  else
173  Serial.println(user_command);
174  Serial.println();
175 
176  switch (user_command)
177  {
178  case 0:
179  Serial.println (F("Enter Frequency(MHz) between 12MHz-80MHZ"));
180  user_command = read_float();
181  if (user_command <= LTC6603_FREQ_MAX && user_command >= LTC6603_FREQ_MIN)
182  freq_settings = user_command;
183  else
184  Serial.println (F("incorrect option"));
185  break;
186  case 1:
187  Serial.println (F("Gain: 0-0dB, 1-6dB, 2-12dB, 3-24dB"));
188  user_command = read_int();
189  switch (user_command)
190  {
191  case 0:
192  control_byte = (control_byte & LTC6603_GAIN_MASK) | LTC6603_GAIN_0dB;
193  break;
194  case 1:
195  control_byte = (control_byte & LTC6603_GAIN_MASK) | LTC6603_GAIN_6dB;
196  break;
197  case 2:
198  control_byte = (control_byte & LTC6603_GAIN_MASK) | LTC6603_GAIN_12dB;
199  break;
200  case 3:
201  control_byte = (control_byte & LTC6603_GAIN_MASK) | LTC6603_GAIN_24dB;
202  break;
203  default:
204  Serial.println (F("incorrect option"));
205  }
206  break;
207  case 2:
208  Serial.println (F("Lowpass Divider: 0-32 1-32 2-128 3-512"));
209  calculate_FreqLow(freq_settings);
210  user_command = read_int();
211  switch (user_command)
212  {
213  case 0:
214  control_byte = (control_byte & LTC6603_LPF_MASK) | LTC6603_LPF3;
215  break;
216  case 1:
217  control_byte = (control_byte & LTC6603_LPF_MASK) | LTC6603_LPF2;
218  break;
219  case 2:
220  control_byte = (control_byte & LTC6603_LPF_MASK) | LTC6603_LPF1;
221  break;
222  case 3:
223  control_byte = (control_byte & LTC6603_LPF_MASK) | LTC6603_LPF0;
224  break;
225  default:
226  Serial.println (F("incorrect option"));
227  }
228  break;
229  case 3:
230  Serial.println (F("No Function Hex Code: 0-0x04, 1-0x08, 2-0x00, 3-0x0C"));
231  user_command = read_int();
232  switch (user_command)
233  {
234  case 0:
235  control_byte = (control_byte & LTC6603_NOFUNC_MASK) | LTC6603_NOFUNC2;
236  break;
237  case 1:
238  control_byte = (control_byte & LTC6603_NOFUNC_MASK) | LTC6603_NOFUNC1;
239  break;
240  case 2:
241  control_byte = (control_byte & LTC6603_NOFUNC_MASK) | LTC6603_NOFUNC0;
242  break;
243  case 3:
244  control_byte = (control_byte & LTC6603_NOFUNC_MASK) | LTC6603_NOFUNC3;
245  break;
246  default:
247  Serial.println (F("incorrect option"));
248  }
249  break;
250  case 4:
251  spi_out = control_byte;
252  LTC6603_write(LTC6603_CS, &spi_out, (uint8_t)1);
253  break;
254  default:
255  Serial.println("Incorrect Option");
256  break;
257  }
258  Serial.print ("\Control byte 0x" + String(control_byte, HEX) + "\n");
259  }
260 }

262 void menu_2_set_gpo()
263 {
264  int16_t user_command; // The user input command
265 
266  while (1)
267  {
268  Serial.print(F("*************************\n\n"));
269  Serial.println(F("0-Set GPIO Low"));

278  Serial.println();

284  LTC6603_write(LTC6603_CS, &spi_out, (uint8_t)1);
285  break;
286  case 1:

290  break;
