65  addr += digitalRead(A3);

69  addr += digitalRead(A1);
70  addr << 1;

72 
73  if (addr > 15) {addr = 15;} // just in case...
74  chaddr = addr2ch[addr];
75  Serial.begin(9600);
76  delay(2);
77  digitalWrite(DE, HIGH);
